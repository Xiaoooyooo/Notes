	****fs****			//读写文件
	fs.readFile('path',callback)	//异步读取文件 同步fs.readFileSync()
	fs.writeFile('xxx.txt',data,callback)	//写文件 同步fs.writeFileSync()
	fs.stat('path',callback)	//获取文件信息
	
	
	****stream****		//流
	var rs = fs.createReadStream('path','utf-8')	//从文件流读取文本内容
	rs.on('data',callback(data))	//data事件表示流的数据已经可以读取
	rs.on('end',callback)		//end事件表示这个流已经到末尾
	rs.on('error',callback)	//error事件表示出错
	
	var ws = fs.createWriteStream('path','utf-8')	//已流的形式写入文件
	ws.write('some text...')	//通过write写入文件
	ws.write('some other text...')	
	ws.end()			//结束写入

	rs.pipe(ws)		//把一个文件流和另一个文件流串起来，
				源文件的所有数据自动写入到目标文件
	rs.pipe(ws,{end:false})	//写入完毕后不关闭ws的write流
